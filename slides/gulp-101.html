Gulp 101

  • npm package, install globally (and locally)
  • streams w/ virtual file format (vinyl)
  • tasks: .src(), .pipe(), .dest()

// general pattern
gulp.task('scripts', function() {
    gulp.src('app/scripts/**/*.js')
        .pipe(anyGulpPlugin())
        .pipe(gulp.dest('dist/scripts'))
        .on('error', errorHandlerFunc);
});
