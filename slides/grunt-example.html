Grunt example

    
module.exports = function(grunt) {

  grunt.initConfig({
    concat: {
        options: {
          separator: ';'
        },
        dist: {
          src: ['scripts/*.js'],
          dest: '<%= distFolder %>/main.js'
        },
        deploy: {
          src:['deploy/*.js'],
          dest:'deploy.js'
        }
    }});
};

grunt.loadNpmTasks('grunt-contrib-concat');

grunt.registerTask('build', ['concat']);
