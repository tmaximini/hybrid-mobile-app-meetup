        // any npm package
        var path = require('path');

        // define tasks
        // example: copy images
        gulp.task('images', function() {
          return gulp.src('app/images/**/*.*')
            .pipe(gulp.dest(path.join('dist', 'images')))
        });

        // ... call this task via 'gulp images'
